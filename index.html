Add this JavaScript to make the counter functional:

// Update counter when navigation buttons are clicked let currentCard = 1; document.querySelector('.carousel-control-next').addEventListener('click', function() { const totalCards = parseInt(document.getElementById('total-cards').textContent); if (currentCard < totalCards) { currentCard++; document.getElementById('current-card').textContent = currentCard; } }); document.querySelector('.carousel-control-prev').addEventListener('click', function() { if (currentCard > 1) { currentCard--; document.getElementById('current-card').textContent = currentCard; } });
